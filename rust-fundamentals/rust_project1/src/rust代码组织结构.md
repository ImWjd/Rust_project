    pub struct Circle {
        pub radius: f64,
    }

    pub enum Shape {
        Circle(Circle),
        Square(f64),
    }

    pub fn area(shape: &Shape) -> f64 {
        match shape {
            Shape::Circle(c) => std::f64::consts::PI * c.radius * c.radius,
            Shape::Square(s) => s * s,
        }
    }
}

fn main() {
    let circle = shapes::Circle { radius: 2.0 };
    let shape = shapes::Shape::Circle(circle);

    println!("Area: {}", shapes::area(&shape));
}
